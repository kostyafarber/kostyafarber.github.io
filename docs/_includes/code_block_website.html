func insertionSort(s string) string {
        sorted := strings.Split(s, "")
        i := 1
        for i < len(s) {
            j := i
            for j > 0 && sorted[j-1] > sorted[j] {
                sorted[j], sorted[j-1] = sorted[j-1], sorted[j]
                j--
            }
            i++
        }
        return strings.Join(sorted, "")
    }
